estate_burghers_monopoly_of_whales = {
	icon = privilege_monopoly_of_goods_whales
	max_absolutism = -10
	loyalty = 0.1
	is_valid = {
		has_dlc = "Emperor"
		any_owned_province = {
			trade_goods = whales
		}
	}
	can_select = {
		custom_trigger_tooltip = {
			tooltip = unavailable_during_revolution_disaster
			NOT = { has_disaster = revolution }
			NOT = {
				has_disaster_progress = {
					disaster = revolution
					value = 1
				}
			}
			NOT = { has_disaster = french_revolution }
			NOT = {
				has_disaster_progress = {
					disaster = french_revolution
					value = 1
				}
			}
		}
	}
	on_granted = {
		add_mercantilism = 1
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				trade_goods = whales
			}
			custom_tooltip = MONOPOLY_INCOME_WHALES_DESC
		}
		custom_tooltip = estate_burghers_monopoly_of_whales_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					trade_goods = whales
				}
				add_province_modifier = {
					name = estate_burghers_monopoly_of_whales_mod
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_burghers_monopoly_of_whales_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_burghers_monopoly_of_whales_mod
				}
				remove_province_modifier = estate_burghers_monopoly_of_whales_mod
			}
		}
	}
	on_invalid = {
		custom_tooltip = estate_burghers_monopoly_of_whales_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_burghers_monopoly_of_whales_mod
				}
				remove_province_modifier = estate_burghers_monopoly_of_whales_mod
			}
		}
	}
	penalties = {
	}
	benefits = {
	}

	cooldown_years = 10
	on_cooldown_expires = {
		country_event = { id = btc_monopoly_privileges.1 }
	}
	ai_will_do = {
		factor = 0
	}
}
