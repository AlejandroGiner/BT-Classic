estate_nobles_monopoly_of_grain = {
	icon = privilege_monopoly_of_goods_grain
	max_absolutism = -10
	loyalty = 0.1
	is_valid = {
		has_dlc = "Emperor"
		any_owned_province = {
			trade_goods = grain
		}
	}
	can_select = {
		custom_trigger_tooltip = {
			tooltip = unavailable_during_revolution_disaster
			NOT = { has_disaster = revolution }
			NOT = {
				has_disaster_progress = {
					disaster = revolution
					value = 1
				}
			}
			NOT = { has_disaster = french_revolution }
			NOT = {
				has_disaster_progress = {
					disaster = french_revolution
					value = 1
				}
			}
		}
	}
	on_granted = {
		add_mercantilism = 1
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				trade_goods = grain
			}
			custom_tooltip = MONOPOLY_INCOME_GRAIN_DESC
		}
		custom_tooltip = estate_nobles_monopoly_of_grain_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					trade_goods = grain
				}
				add_province_modifier = {
					name = estate_nobles_monopoly_of_grain_mod
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_nobles_monopoly_of_grain_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_nobles_monopoly_of_grain_mod
				}
				remove_province_modifier = estate_nobles_monopoly_of_grain_mod
			}
		}
	}
	on_invalid = {
		custom_tooltip = estate_nobles_monopoly_of_grain_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_nobles_monopoly_of_grain_mod
				}
				remove_province_modifier = estate_nobles_monopoly_of_grain_mod
			}
		}
	}
	penalties = {
	}
	benefits = {
	}
	cooldown_years = 10
	on_cooldown_expires = {
		country_event = { id = monopoly_privileges.51 }
	}
	ai_will_do = {
		factor = 0
	}
}
