mamluk_succession_disaster = {
	potential = {
		tag = MAM
		has_reform = mamluk_government
		is_free_or_tributary_trigger = yes
	}
	
	can_start = {
		has_any_disaster = no
		has_reform = mamluk_government
		NOT = {
			months_of_ruling = 24
			legitimacy = 60
			ruler_has_personality = righteous_personality
			ruler_has_personality = silver_tongue_personality
			ruler_has_personality = charismatic_negotiator_personality
			ruler_has_personality = fierce_negotiator_personality
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			AND = {
				legitimacy = 80
				stability = 2
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 10
			NOT = { legitimacy = 80 }
		}
		modifier = {
			factor = 5
			NOT = { adm = 3 }
		}
		modifier = {
			factor = 5
			NOT = { dip = 3 }
		}
	}
	
	can_end = {
		OR = {
			custom_trigger_tooltip = {
				tooltip = disaster_active_for_7_years_tooltip
				had_country_flag = { flag = mamluk_succession_started days = 2555 #7 years }
			}
			NOT = {
				num_of_rebel_armies = 1
				num_of_rebel_controlled_provinces = 1
			}
		}
	}
	
	modifier = {
		stability_cost_modifier = 0.5
		global_unrest = 5
		legitimacy = -1
		land_morale = -0.1
	}
	
	on_start = btc_mamluk_succession_disaster.1
	on_end = btc_mamluk_succession_disaster.2
	on_monthly = {
		events = {
		
		}
		random_events = {
		
		}
	}
}
