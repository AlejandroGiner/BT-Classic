	pru_conquer_silesia = {
		icon = mission_assemble_an_army
		required_missions = { connect_the_prussian_lands }

			OR = {
				area = legnica_area

			}

			}

				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_country_modifier = {
				name = "prussian_ambition"
				duration = 7300
			}
			
			#Copying from Austria to add PU CB to Cleves (& Julich if Cleves does not exist). This was a crownland, after all.
			if = {
				limit = {
					exists = KLE
					is_subject = no
					KLE = {
						is_subject = no
					}
				}
				add_casus_belli = {
					type = cb_restore_personal_union
					months = 300
					target = KLE
				}
			}
			else_if = {
				limit = {
					exists = BRG
					is_subject = no
					BRG = {
						is_subject = no
					}
				}
				add_casus_belli = {
					type = cb_restore_personal_union
					months = 300
					target = BRG
				}
			}
			else = {
				#Claims
				custom_tooltip = pru_duchy_of_cleves_claims
				hidden_effect = {
					every_province = {
						limit = {
							pru_duchy_of_cleves_provinces = yes
							NOT = { is_core = ROOT }
							NOT = { owned_by = ROOT }
						}
						add_permanent_claim = ROOT
					}
				}
			}
		}
	}
