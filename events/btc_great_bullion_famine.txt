namespace = bullion

#Start of the Great Bullion Famine
country_event = {
	id = bullion.0
	title = bullion.0.t
	desc = bullion.0.d
	picture = OVEREXTENSION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		capital_scope = {
			superregion = europe_superregion
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						tag = MAM
						tag = TUR
						AND = {
							religion_group = christian
							capital_scope = {
								continent = europe
							}
						}
					}
					NOT = { tag = ROOT }
				}
				country_event = { id = bullion.0 }
			}
		}
	}
	
	option = {
		name = bullion.0.a
		trigger = {
			OR = {
				tag = MAM
				tag = TUR
			}
		}
		add_country_modifier = {
			name = east_west_profit_trade
			duration = -1
		}
	}
	option = {
		name = bullion.0.b
		trigger = {
			NOT = {
				tag = MAM
				tag = TUR
			}
		}
		add_country_modifier = {
			name = east_west_deficit_trade
			duration = -1
		}
	}
}

#North African Gold

	id = bullion.2
	title = bullion.2.t
	desc = bullion.2.d

		NOT = { has_global_flag = great_bullion_famine_ended }

		}
		NOT = {
			tag = MAM
			tag = TUR
		}
