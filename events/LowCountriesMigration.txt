country_event = { #Fall of power of the staple ports
	id = btc_lc_cities.5
	title = btc_lc_cities.5.t
	desc = btc_lc_cities.5.d
	picture = TRADE_GOODS_FURS_FISH_AND_SALT_eventPicture
	
	trigger = {
		is_year = 1600
		any_owned_province = {
			OR = {
				has_province_modifier = english_staple_port
				has_province_modifier = scottish_staple_port
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
	}
	
	option = { #Favored exportation of Raw Materials
		name = btc_lc_cities.5.a
		every_owned_province = {
			if = {
				limit = {has_province_modifier = english_staple_port}
				remove_province_modifier = english_staple_port
				add_base_production = 2
				if = {
					limit = {province_id = 97} #Amsterdam
					change_trade_goods = cloth
				}
			}
			else_if = {
				limit = {has_province_modifier = scottish_staple_port}
				remove_province_modifier = scottish_staple_port
				add_base_production = 2
				if = {
					limit = {province_id = 97} #Amsterdam
					change_trade_goods = cloth
				}
			}
		}
	}
}
