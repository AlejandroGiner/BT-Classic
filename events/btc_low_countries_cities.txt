country_event = { #Fall of power of the staple ports
	id = btc_lc_cities.5
	title = btc_lc_cities.5.t
	desc = btc_lc_cities.5.d
	
	trigger = {
		is_year = 1600
		any_owned_province = {
			limit = {
				OR = {
					has_province_modifier = english_staple_port
					has_province_modifier = scottish_staple_port
				}
			}
		}
	}
	mean_time_to_happen = {
		months = 100
	}
	
	option = {
		name = btc_lc_cities.5.a
		any_owned_province = {
			limit = {
				OR = {
					has_province_modifier = english_staple_port
					has_province_modifier = scottish_staple_port
				}
			}
			remove_province_modifier = english_staple_port
			remove_province_modifier = scottish_staple_port
			THIS = {add_base_production = 2}
		}
	}
}
