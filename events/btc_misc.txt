#Siege defended with bees
#ROOT = attacker
country_event = {
	id = btc_misc.5
	title = btc_misc.5.t
	desc = btc_misc.5.d
	picture = SIEGE_eventPicture #placeholder
  
	trigger = {
		any_known_country = {
			war_with = ROOT
			any_owned_province = {
				sieged_by = ROOT
				fort_level = 1
				trade_goods = apiculture
			}
		}
	}
  
	mean_time_to_happen = {
		years = 50
	}

	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					war_with = ROOT
					any_owned_province = {
						sieged_by = ROOT
						fort_level = 1
						trade_goods = apiculture
					}
				}
				save_event_target_as = bee_defender
				random_owned_province = {
					limit = {
						sieged_by = ROOT
						fort_level = 1
						trade_goods = Apiculture
					}
					save_event_target_as = bee_province
				}
			}
		}
	}
  
	option = {
		name = btc_misc.5.a
		change_siege = -2
		kill_units = {
			who = ROOT
			amount = 1
		}
	}
}
