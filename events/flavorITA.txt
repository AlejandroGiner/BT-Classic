namespace = flavor_ita
normal_or_historical_nations = yes

country_event = {
  id =  flavor_ita.1
  title = "flavor_ita.1.title"
  desc =  "flavor_ita.1.desc"
  picture = COASTLINE_MAP_eventPicture

  fire_only_once = yes

  trigger = {
    tag = ITA
    NOT = { primary_culture = tuscan }
    NOT = { primary_culture = umbrian }
    NOT = { has_country_flag = italy_tuscan_language_flag }
  }

  mean_time_to_happen = {
    days = 1
  }

  immediate = {
    set_country_flag = italy_tuscan_language_flag
  }

  option = {  #adopt tuscan language
    name = flavor_ita.1.a
    add_country_modifier = {
      name = btc_language_of_italy
      duration = 7300
    }
    every_owned_province = {
      limit = { has_owner_culture = yes }
      add_province_modifier = {
        name = btc_italy_local_language_unrest
        duration = 1825
      }
    }
    change_primary_culture = tuscan
    ai_chance = { factor = 50 }
  }

  option = {  #keep local language
    name = flavor_ita.1.b
    add_prestige = -5
    ai_chance = { factor = 50 }
  }
}
